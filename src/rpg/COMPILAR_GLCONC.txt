# --- 0. Compilacion completa (recomendado tras cada cambio de fuentes) ---
# Subir src/rpg/GLCONBUI.CLLE a QTALLERCP1 (miembro GLCONBUI, tipo CLLE)
# CRTCLPGM PGM(FMRAMIREZ1/GLCONBUI) SRCFILE(FMRAMIREZ1/QTALLERCP1) SRCMBR(GLCONBUI)
# CALL PGM(FMRAMIREZ1/GLCONBUI) PARM('FMRAMIREZ1')
#
# GLCONBUI ejecuta DLTSPLF FILE(*SELECT) al inicio, compila en orden y
# detiene el job ante el primer error (sin MONMSG que oculte fallos).
# El ultimo SNDPGMMSG indica el fuente/paso en curso; el spool nuevo muestra el detalle.
# --- 1. Copybooks compartidos (no se compilan solos) ---
# Subir a QTALLERCP1 antes de compilar modulos:
#   GLCONH  - constantes C_* y plantillas T_* (Documentacion_IBMi/estandar_codigo_rpg.md)
#   GLCONPR  - prototipos ExtProc para GLRPT001 (despues de GLCONH)
#   GLCONPRU - prototipos servicio GLUTIL001 (GLOUT001 y otros NOMAIN)
#
# GLDAT001 usa likeds de plantillas GLCONH en parametros y locales de proc;
# no requiere estructuras wrk* duplicadas en el copybook.
#
# Reglas prototipos Dcl-Pr con ExtProc:
#   - Con parametros: retorno ANTES de ExtProc (int(10) ExtProc('DatLoadPartidas'))
#   - ExtProc en MAYUSCULAS (export SRVPGM: DATINITWORKTABLES, no DatInitWorkTables)
#   - No poner tipo despues de ExtProc (RNF3742); no usar Dcl-Pi/End-Pi
#   - Tras cambiar GLCONPR: recompilar GLRPT001 antes de CRTPGM
#   - Formato libre normal desde columna 1, sin tabular a col 6
#   - Encabezados // permitidos en miembros **FREE
#   - SQLRPGLE: /COPY + RPGPPOPT LV2 en CRTSQLRPGI
#   - Ver: https://www.nicklitten.com/what-is-the-difference-between-copy-and-include-in-sql-rpgle/
#
# GLDAT001: /copy QTALLERCP1,GLCONH; host SQL via likeds en parms y locales
#
# Parametros comunes SQLRPGLE:
#   RPGPPOPT(*LVL2)           - preprocesador RPG antes del precompilador SQL
#   OPTION(*XREF *SECLVL)     - cross-reference SQL + texto nivel 2 (reason codes)
#   COMPILEOPT('OPTION(*SECLVL *XREF)') - listado RPG de la segunda fase
#   OUTPUT(*PRINT)            - opcional: listado completo del precompilador SQL
#
# Parametros comunes RPGLE puro (CRTRPGMOD):
#   Sin RPGPPOPT (solo aplica a CRTSQLRPGI / precompilador SQL)
#   /COPY se expande nativamente al compilar el modulo
#   OPTION(*SRCSTMT *SECLVL *XREF) - alineado con Ctl-Opt Option(*SrcStmt)

# --- 2. Programa de servicio de utilidades (GLUTIL001.SQLRPGLE) ---
CRTSQLRPGI OBJ(FMRAMIREZ1/GLUTIL001) SRCFILE(FMRAMIREZ1/QTALLERCP1) SRCMBR(GLUTIL001)
  OBJTYPE(*MODULE) COMMIT(*NONE) DBGVIEW(*SOURCE)
  RPGPPOPT(*LVL2) OPTION(*XREF *SECLVL)
  COMPILEOPT('OPTION(*SECLVL *XREF)')
  TEXT('Modulo utilidades conciliacion')

CRTSRVPGM SRVPGM(FMRAMIREZ1/GLUTIL001) MODULE(FMRAMIREZ1/GLUTIL001)
  EXPORT(*ALL) TEXT('Servicio utilidades conciliacion GLBLN')

# --- 3. Modulos ILE (OBJTYPE *MODULE) ---
CRTSQLRPGI OBJ(FMRAMIREZ1/GLDAT001) SRCFILE(FMRAMIREZ1/QTALLERCP1) SRCMBR(GLDAT001)
  OBJTYPE(*MODULE) COMMIT(*NONE) DBGVIEW(*SOURCE)
  RPGPPOPT(*LVL2) OPTION(*XREF *SECLVL)
  COMPILEOPT('OPTION(*SECLVL *XREF)')
  TEXT('Modulo acceso datos GLBLN')

CRTRPGMOD MODULE(FMRAMIREZ1/GLNEG001) SRCFILE(FMRAMIREZ1/QTALLERCP1) SRCMBR(GLNEG001)
  DBGVIEW(*SOURCE) OPTION(*SRCSTMT *SECLVL *XREF)
  TEXT('Modulo reglas negocio conciliacion')

CRTSQLRPGI OBJ(FMRAMIREZ1/GLOUT001) SRCFILE(FMRAMIREZ1/QTALLERCP1) SRCMBR(GLOUT001)
  OBJTYPE(*MODULE) COMMIT(*NONE) DBGVIEW(*SOURCE)
  RPGPPOPT(*LVL2) OPTION(*XREF *SECLVL)
  COMPILEOPT('OPTION(*SECLVL *XREF)')
  TEXT('Modulo salida JSON IFS')

# --- 4. Programas de servicio de modulos exportadores ---
# Tras cambiar fuentes o GLCONPR: recompilar *MODULE y repetir CRTSRVPGM
CRTSRVPGM SRVPGM(FMRAMIREZ1/GLDAT001) MODULE(FMRAMIREZ1/GLDAT001)
  EXPORT(*ALL) ACTGRP(*CALLER) TEXT('Servicio datos GLBLN')

CRTSRVPGM SRVPGM(FMRAMIREZ1/GLNEG001) MODULE(FMRAMIREZ1/GLNEG001)
  BNDSRVPGM(FMRAMIREZ1/GLUTIL001) EXPORT(*ALL) ACTGRP(*CALLER)
  TEXT('Servicio negocio GLBLN')

# GLOUT001 NO es SRVPGM; recompilar modulo tras cambiar GLCONPRU (WRITEIFSUTF8)
CRTSQLRPGI OBJ(FMRAMIREZ1/GLOUT001) SRCFILE(FMRAMIREZ1/QTALLERCP1) SRCMBR(GLOUT001)
  OBJTYPE(*MODULE) COMMIT(*NONE) DBGVIEW(*SOURCE)
  RPGPPOPT(*LVL2) OPTION(*XREF *SECLVL)
  COMPILEOPT('OPTION(*SECLVL *XREF)')
  TEXT('Modulo salida JSON IFS')

# --- 5. Programa principal (modulo + enlace) ---
# IMPORTANTE: resubir GLCONPR y recompilar GLRPT001 si cambian prototipos
# ACTGRP(*CALLER) en CRTPGM; DFTACTGRP no es parametro valido de CRTPGM
CRTSQLRPGI OBJ(FMRAMIREZ1/GLRPT001) SRCFILE(FMRAMIREZ1/QTALLERCP1) SRCMBR(GLRPT001)
  OBJTYPE(*MODULE) COMMIT(*NONE) DBGVIEW(*SOURCE)
  RPGPPOPT(*LVL2) OPTION(*XREF *SECLVL)
  COMPILEOPT('OPTION(*SECLVL *XREF)')
  TEXT('Modulo principal conciliacion GLBLN')

# GLOUT001 *MODULE en MODULE(); GLDAT/GLNEG/GLUTIL como SRVPGM
CRTPGM PGM(FMRAMIREZ1/GLRPT001) MODULE(FMRAMIREZ1/GLRPT001 FMRAMIREZ1/GLOUT001)
  BNDSRVPGM(FMRAMIREZ1/GLDAT001 FMRAMIREZ1/GLNEG001 FMRAMIREZ1/GLUTIL001)
  ACTGRP(*CALLER) TEXT('Conciliacion GLBLN')

# --- 6. Diagnostico de enlace (CPD5D02 / CPF5D04) ---
# CPF5D04 "Not authorized to QYPSSRVS in QSYS" es RUIDO del enlazador IBM:
#   al no resolver un simbolo, busca en BNDDIR del sistema (QUSAPIBD) y
#   reporta falta de autoridad en SRVPGM restringidos. IGNORAR ese mensaje.
#   Enfocarse en CPD5D02 (simbolo real sin definicion).
#   Ref: https://www.ibm.com/support/pages/explanation-error-messages-cpd5d02-and-cpf5d04-when-creating-program-or-service-program
#
# Verificar que existen *MODULE (no *PGM) antes de CRTPGM:
#   WRKOBJ OBJ(FMRAMIREZ1/GLDAT001) OBJTYPE(*MODULE)
#   WRKOBJ OBJ(FMRAMIREZ1/GLNEG001) OBJTYPE(*MODULE)
#   WRKOBJ OBJ(FMRAMIREZ1/GLOUT001) OBJTYPE(*MODULE)
#   WRKOBJ OBJ(FMRAMIREZ1/GLRPT001) OBJTYPE(*MODULE)
#   WRKOBJ OBJ(FMRAMIREZ1/GLUTIL001) OBJTYPE(*SRVPGM)
#
# Verificar exports antes de CRTPGM:
#   DSPSRVPGM SRVPGM(FMRAMIREZ1/GLNEG001) DETAIL(*PROCEXP)
#     -> debe incluir NEGGENHASHCUENTA (NegGenHashCuenta, dos G)
#   DSPSRVPGM SRVPGM(FMRAMIREZ1/GLUTIL001) DETAIL(*PROCEXP)
#     -> debe incluir UTILHASHMD5HEX y WRITEIFSUTF8
#   DSPMOD MODULE(FMRAMIREZ1/GLOUT001) DETAIL(*IMPORT)
#     -> import WRITEIFSUTF8 (no WriteIfsUtf8 mixto; recompilar si aparece mixto)
#
# Si EXPORT esta vacio: recompilar con OBJTYPE(*MODULE), no *PGM.
# Si hay *PGM viejo con el mismo nombre, borrarlo antes de recompilar:
#   DLTPGM PGM(FMRAMIREZ1/GLDAT001)
#   CRTSQLRPGI OBJ(FMRAMIREZ1/GLDAT001) ... OBJTYPE(*MODULE) ...
